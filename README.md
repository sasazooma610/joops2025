	<script>
    document.querySelectorAll('.progress-text').forEach(progressText => {
        // الحصول على القيم من السمات
        const appliedCount = parseInt(progressText.getAttribute('data-applied'));
        const capacity = parseInt(progressText.getAttribute('data-capacity'));
        
        // حساب النسبة
        const percentage = (appliedCount / capacity) * 100;
        
        // تحديث عرض المؤشر والنص
        const progressBar = progressText.previousElementSibling.querySelector('.progress');
        progressBar.style.width = percentage + "%";
        progressText.querySelector('.applied-text').innerText = appliedCount;
        progressText.querySelector('.capacity-value').innerText = capacity;
    });
</script>
